	ls foulder2/   //посмотреть содержимое папки foulder2
	grep -R first foulder2/    //поиск нужного паттерна first в директории foulder2, покажет все файлы в которых есть искомый паттерн
	grep -Ri FirsT foulder2/   //-i поиск first игнорируя регистр букв по всем файлам в директории foulder2
	grep -Rcv first foulder2/  //-cv подсчитает и покажет кол-во строк, которые не совпадают с нашим паттерном по всем файлам в папке
	grep -Rс first foulder2/   // -c подсчитает и покажет кол-во строк, которые совпадают с нашим паттерном по всем файлам в папке

	grep -v first f1.txt       //-v показать несовпавшие строки
